### –ê. –ê–ª–≥–æ—Ä–∏—Ç–º –ü—Ä–∏–º–∞ (Prim)

**–ò–¥–µ—è:** –í—ã—Ä–∞—â–∏–≤–∞–µ–º "–ø–ª–µ—Å–µ–Ω—å" –∏–ª–∏ "–ø—è—Ç–Ω–æ". –ù–∞—á–∏–Ω–∞–µ–º —Å –æ–¥–Ω–æ–π –≤–µ—Ä—à–∏–Ω—ã –∏ –≤—Å–µ–≥–¥–∞ –∂–∞–¥–Ω–æ –∑–∞—Ö–≤–∞—Ç—ã–≤–∞–µ–º —Å–∞–º–æ–µ –±–ª–∏–∑–∫–æ–µ –Ω–µ –ø–æ—Å–µ—â–µ–Ω–Ω–æ–µ —Ä–µ–±—Ä–æ, –∏—Å—Ö–æ–¥—è—â–µ–µ –∏–∑ –Ω–∞—à–µ–≥–æ –ø—è—Ç–Ω–∞.
**–°—Ç—Ä—É–∫—Ç—É—Ä–∞:** Priority Queue.

### –ë. –ê–ª–≥–æ—Ä–∏—Ç–º –ö—Ä—É—Å–∫–∞–ª–∞ (Kruskal)

**–ò–¥–µ—è:** –õ–µ—Å, –∫–æ—Ç–æ—Ä—ã–π —Å—Ä–∞—Å—Ç–∞–µ—Ç—Å—è. –°–æ—Ä—Ç–∏—Ä—É–µ–º **–≤—Å–µ** —Ä–µ–±—Ä–∞ –æ—Ç –¥–µ—à–µ–≤—ã—Ö –∫ –¥–æ—Ä–æ–≥–∏–º. –ë–µ—Ä–µ–º —Ä–µ–±—Ä–æ, –µ—Å–ª–∏ –æ–Ω–æ —Å–æ–µ–¥–∏–Ω—è–µ—Ç –¥–≤–∞ —Ä–∞–∑–Ω—ã—Ö –¥–µ—Ä–µ–≤–∞ (–Ω–µ—Ç —Ü–∏–∫–ª–∞).
**–°—Ç—Ä—É–∫—Ç—É—Ä–∞:** DSU (Disjoint Set Union / –°–ù–ú).

### üíª –ö–æ–¥ C++ (–ü—Ä–∏–º ‚Äî –æ–ø—Ç–∏–º–∞–ª–µ–Ω –¥–ª—è –ø–ª–æ—Ç–Ω—ã—Ö –≥—Ä–∞—Ñ–æ–≤)

```cpp
#include <queue>

// {weight, vertex}
typedef pair<int, int> pii; 

int primMST(int n, vector<vector<pii>>& adj) {
    priority_queue<pii, vector<pii>, greater<pii>> pq;
    
    int src = 0; // –ù–∞—á–∏–Ω–∞–µ–º —Å –≤–µ—Ä—à–∏–Ω—ã 0
    vector<int> key(n, INT_MAX); // –ú–∏–Ω–∏–º–∞–ª—å–Ω—ã–π –≤–µ—Å —Ä–µ–±—Ä–∞ –¥–ª—è –≤—Ö–æ–¥–∞ –≤ –≤–µ—Ä—à–∏–Ω—É
    vector<bool> inMST(n, false); // –í–∫–ª—é—á–µ–Ω–∞ –ª–∏ –≤–µ—Ä—à–∏–Ω–∞ –≤ –æ—Å—Ç–æ–≤

    pq.push({0, src});
    key[src] = 0;
    
    int mst_weight = 0;

    while (!pq.empty()) {
        int u = pq.top().second;
        pq.pop();

        if (inMST[u]) continue; // –ï—Å–ª–∏ —É–∂–µ –¥–æ–±–∞–≤–∏–ª–∏, –ø—Ä–æ–ø—É—Å–∫–∞–µ–º
        inMST[u] = true;
        mst_weight += key[u]; // –î–æ–±–∞–≤–ª—è–µ–º –≤–µ—Å —Ä–µ–±—Ä–∞, –ø–æ –∫–æ—Ç–æ—Ä–æ–º—É –ø—Ä–∏—à–ª–∏

        for (auto& edge : adj[u]) {
            int v = edge.second;
            int weight = edge.first;

            // –ï—Å–ª–∏ v –µ—â–µ –Ω–µ –≤ MST –∏ –Ω–æ–≤–æ–µ —Ä–µ–±—Ä–æ –∫–æ—Ä–æ—á–µ —Å—Ç–∞—Ä–æ–≥–æ –∏–∑–≤–µ—Å—Ç–Ω–æ–≥–æ –ø—É—Ç–∏
            if (!inMST[v] && weight < key[v]) {
                key[v] = weight;
                pq.push({key[v], v});
            }
        }
    }
    return mst_weight;
}
```

## 5\. Shortest Path (Dijkstra)

### üß† –¢–µ–æ—Ä–∏—è

–ù–∞–π—Ç–∏ –∫—Ä–∞—Ç—á–∞–π—à–∏–π –ø—É—Ç—å –æ—Ç —Ç–æ—á–∫–∏ –ê –¥–æ –≤—Å–µ—Ö –æ—Å—Ç–∞–ª—å–Ω—ã—Ö. –†–∞–±–æ—Ç–∞–µ—Ç —Ç–æ–ª—å–∫–æ —Å **–Ω–µ–æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω—ã–º–∏** –≤–µ—Å–∞–º–∏.

**–°—É—Ç—å (–†–µ–ª–∞–∫—Å–∞—Ü–∏—è):**
–ï—Å–ª–∏ —è –∑–Ω–∞—é, —á—Ç–æ –¥–æ –≥–æ—Ä–æ–¥–∞ $B$ –µ—Ö–∞—Ç—å 100–∫–º, –∞ –æ—Ç $B$ –¥–æ $C$ ‚Äî 10–∫–º, —Ç–æ –ø—É—Ç—å –¥–æ $C$ —á–µ—Ä–µ–∑ $B$ —Ä–∞–≤–µ–Ω 110–∫–º. –ï—Å–ª–∏ —è —Ä–∞–Ω—å—à–µ –¥—É–º–∞–ª, —á—Ç–æ –¥–æ $C$ –µ—Ö–∞—Ç—å 150–∫–º, —è "—Ä–∞—Å—Å–ª–∞–±–ª—è—é" (—É–ª—É—á—à–∞—é) —ç—Ç–æ—Ç –ø—É—Ç—å –¥–æ 110.

### üëÄ –í–∏–∑—É–∞–ª–∏–∑–∞—Ü–∏—è

–ü—Ä–µ–¥—Å—Ç–∞–≤—å –≤–æ–¥—É, –∫–æ—Ç–æ—Ä–∞—è —Ç–µ—á–µ—Ç –ø–æ —Ç—Ä—É–±–∞–º —Ä–∞–∑–Ω–æ–π —à–∏—Ä–∏–Ω—ã (–≤–µ—Å —Ä–µ–±—Ä–∞). –í–æ–¥–∞ –¥–æ–π–¥–µ—Ç –¥–æ –±–ª–∏–∂–∞–π—à–∏—Ö —É–∑–ª–æ–≤ –±—ã—Å—Ç—Ä–µ–µ. –î–µ–π–∫—Å—Ç—Ä–∞ –≥–∞—Ä–∞–Ω—Ç–∏—Ä—É–µ—Ç, —á—Ç–æ –∫–∞–∫ —Ç–æ–ª—å–∫–æ –º—ã "–∑–∞–∫—Ä—ã–≤–∞–µ–º" –≤–µ—Ä—à–∏–Ω—É (–¥–æ—Å—Ç–∞–µ–º –∏–∑ –æ—á–µ—Ä–µ–¥–∏), –º—ã –Ω–∞—à–ª–∏ –¥–æ –Ω–µ—ë –∫—Ä–∞—Ç—á–∞–π—à–∏–π –ø—É—Ç—å –Ω–∞–≤—Å–µ–≥–¥–∞.

### üíª –ö–æ–¥ C++

```cpp
const int INF = 1e9;

void dijkstra(int start, int n, vector<vector<pii>>& adj) {
    vector<int> dist(n, INF);
    dist[start] = 0;
    
    // min-priority queue: {distance, vertex}
    priority_queue<pii, vector<pii>, greater<pii>> pq;
    pq.push({0, start});

    while (!pq.empty()) {
        int d = pq.top().first;
        int u = pq.top().second;
        pq.pop();

        // –í–∞–∂–Ω–∞—è –æ–ø—Ç–∏–º–∏–∑–∞—Ü–∏—è: –µ—Å–ª–∏ –º—ã –¥–æ—Å—Ç–∞–ª–∏ "—Å—Ç–∞—Ä—É—é" –∑–∞–ø–∏—Å—å —Å –±–æ–ª—å—à–∏–º —Ä–∞—Å—Å—Ç–æ—è–Ω–∏–µ–º
        if (d > dist[u]) continue;

        for (auto& edge : adj[u]) {
            int v = edge.second;
            int weight = edge.first;

            // –†–µ–ª–∞–∫—Å–∞—Ü–∏—è
            if (dist[u] + weight < dist[v]) {
                dist[v] = dist[u] + weight;
                pq.push({dist[v], v});
            }
        }
    }

    // –í—ã–≤–æ–¥ —Ä–µ–∑—É–ª—å—Ç–∞—Ç–æ–≤
    for (int i = 0; i < n; ++i) {
        cout << "Dist to " << i << ": " << dist[i] << endl;
    }
}
```
